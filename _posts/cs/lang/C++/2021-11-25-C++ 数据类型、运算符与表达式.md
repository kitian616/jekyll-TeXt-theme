# C++ 常量

```c++
#include <iostream>

using namespace std;
/**
 * C++ 常量
 * 整型常量：
 * 十进制：2010， 100
 * 八进制：0123，020（由0开始，后面只能接0-7）
 * 十六进制：0xff, 0x12（由0x开始，后面能跟0-9,a-f/A-F)
 *
 * 浮点常量：
 * 0.23, 2.34
 * 2.3e12, 0.23E-4
 *
 * 字符常量：
 * 普通字符：'a', '1'
 * 普通转义字符：'\\'(一个\), '\n'(换行)
 * 8/16进制转义字符：'\013', '\x12'
 * 
 * 字符串常量：
 * "abcdefg", "china\n"
 * 
 * 布尔常量：
 * true, false
 *
 */
int main() {
    int a = 1, b = 2, c;
    c = a + b;
    cout << 123 << endl;    // 123
    cout << 0123 << endl;   // 83
    cout << 0xff << endl;   // 255
    cout << 0xFF << endl;   // 255
    cout << 2.3*1000000000000 << endl;   // 2.3e+12
    cout << 0.0000000000023 << endl;   // 2.3e-12
    return 0;
}
```

# 类型转换

```c++
#include <iostream>

using namespace std;

/**
 * 类型转换
 * 显式类型转换：
 * (int)3.14 将 3.14 转换为整数3，不是四舍五入，而是截断
 * (double)3 将 3 转换成双精度的浮点数 3.0
 * 
 * 隐式类型转换：提升规则
 * 表达式中运算符的操作数类型与要求不符时，进行隐式类型转换
 * 隐式类型转换的规则：将不同类型的数据转换成精度最高、占用内存最多的那个数据的类型
 * 例子： 2 + 3.4 -> 2.0 + 3.4
 * (double)3 / 2 -> 3.0 / 2 (类型转换的优先级更高)
 * (double) (3 / 2) -> (double) 1 -> 1.0
 */

int main() {
    cout << (int) 3.14 << endl; // 3
    cout << (int) 3.94 << endl; // 3

    double x = 3.96;
    cout << (int) x << endl;    // 这个类型转换并没有影响原来的数据，而是生成了新的无名临时变量
    cout << x << endl;
    return 0;
}
```

# 运算符与表达式

```c++
#include <iostream>

using namespace std;

/**
 * 运算符与表达式
 * 算数运算符：
 *  一元：
 *      + - ++(前自增、后自增) --(前自减、后自减) 从汇编代码的角度去理解自增自减会好理解很多 自增指令 INC 可以在赋值前，也可以在之后
 *      int a, b = 3;
 *      a = ++b; // a = 4, b = 4
 *      a = b++; // a = 4, b = 5
 *      自增运算符如果在一个表达式中使用过多，很容易产生歧义，降低代码可读性
 *      而且由于编译器的优化，其结果可能并没有保证
 *
 *  二元：
 *      + - * / (两个整数相除，结果为整数，小数部分直接舍去)
 *      % 求余，正数没有疑义，但是对于负数，可能不同的编译器有不同的选择
 *
 *  关系运算：
 *      < > <= >= == != 都是双目运算符(注意区分相等符号和赋值符号)
 *      a > b   a != 3
 *
 *  逻辑运算：
 *      !(逻辑非)、||(逻辑或)、&&(逻辑与)
 *      !(a > b)
 *      (a >= 0) && (a <= 100)
 *      对于写法：0 <= a <= 100，在C++中，会先比较 0 <= a, 因为C++中关系运算返回 1 或 0，所以会用 1 或 0 继续与100比较
 *      显然这种用法是不可取的
 *
 *  位运算符：
 *      运算符：
 *      ~（按位取非）、&（按位与）、｜（按位或）
 *      ^（按位异或）、<<（左移位）、>>（右移位）
 *      12&6 = 4：
 *          12  0000 1100
 *           6  0000 0110
 *         = 4  0000 0100
 *         
 *  赋值运算符:
 *      =
 *      *=  /=  %=  +=  -=  >>=  <<=  &=  ^=  |=
 *      
 *  条件运算符：
 *      三目运算符，优先级大于赋值运算符：?:
 *      a = (x > y) ? x : y;    // 如果 x > y, a = x; 否则 a = y
 */

int main() {
    int a, b = 3;
    a = b++;
    cout << a << endl;  // a = 3

    a = ++b;
    cout << a << endl;  // a = 5
    // 21 对于我的编译器而言，是从左到右依次计算的 6 + 7 + 8 = 21
    // 也有些编译器会扫描整个表达式让b自增3次，然后 b * 3 = 24
    a = ++b + ++b + ++b;
    cout << a << endl;

    // 以下结果是对我的编译器而言的
    // 余数的符号与被求数的符号相同
    cout << 17 % 3 << endl;     // 2
    cout << 17 % -3 << endl;    // 2
    cout << -17 % 3 << endl;    // -2
    cout << -17 % -3 << endl;   // -2

    int c = 4;
//    cout << c == 3 << endl; // 这里 << 流操作运算符的优先级高于 ==，所以这里会报错
    cout << (c == 3) << endl; // 0

    // 以下情况，不管c的值是多少，结果都是1
    cout << (0 <= c <= 100) << endl;    // 1
    c = 400;
    cout << (0 <= c <= 100) << endl;    // 1

    cout << (12 & 6) << endl;   // 4
    cout << (12 | 6) << endl;   // 14
    cout << (~12) << endl;      // -13
    cout << (12 << 1) << endl;      // 24
    cout << (12 >> 1) << endl;      // 6

    return 0;
}
```

# 练习一

```c++
#include <iostream>

using namespace std;

/**
 * 编写程序，输入一个三位数，计算并输出该三位数的3个数字之和。如输入：123，则输出：6
 */

int main() {
    int num;
    cin >> num;
    int sum = 0;
    while (num != 0) {
        sum += num % 10;
        num /= 10;
    }

    cout << sum <<endl;
}
```

# 练习二

```c++
#include <iostream>

using namespace std;

/**
 * 编写程序，输入一个正整数，输出其二进制表示的倒数第3位（最低位为倒数第1位）。
 * 如输入：12，则输出：1
 *   输入：8，则输出：0
 */

int main() {
    int num;
    cin >> num;
    // 与 0100 = 4 相与，即可得到 num 倒数第三位的值，向右移两位即是结果
    int result = (num & 4) >> 2;
    cout << result << endl;
}
```
