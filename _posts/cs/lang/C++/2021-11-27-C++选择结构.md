# if 选择结构

```c++
#include <iostream>

using namespace std;

/**
 * if 语句:
 *      实现单路、双路、多路选择
 * 一个 if 分支或 else 分支有且只有一条语句
 * 如果某个分支不需要做任何事怎么办？
 *      使用空语句
 * 如果某分支需要多条语句怎么办？
 *      使用花括号 {} 组成复合语句
 *      建议只有单条语句时也用花括号 {} 括起来从而提高代码的可读性。
 */

int main() {
    int score = 0;
    // 单路选择
    if (score >= 60) {
        cout << "Passed";
    }
    
    // 双路选择
    int s;
    cin >> s;
    if (s == 1) {
        cout << "male";
    } else {
        cout << "female";
    }
    
    // 多路选择：嵌套
    int y;
    cin >> y;
    double rate;
    if (y == 1) {
        rate = 0.02;
    } else if (y == 2) {
        rate = 0.03;
    } else {
        rate = 0.01;
    }
    cout << rate;

    int speed;
    cin >> speed;
    if (speed > 200) ;  // do nothing
    else {
        cout << "超速";
    }

    int a, b, c;
    // 多条语句时用花括号 {} 括起来
    if (a > b) {
        // 交换 a 和 b 的值
        c = a;
        a = b;
        b = c;
    }

    if (a > b)
        c = a;  // if 语句只包含这一句
    a = b;
    b = c;
}
```

# 浮点数相等问题

```c++
#include <iostream>
#include <cmath>

using namespace std;

/**
 * 例子：
 * 浮点数相等问题
 * 判断两数之差的绝对值是否小于一个足够小的数，如 1e-12
 */

int main() {

    double a = 0.19, b = 0.57;  // 内存中存的值： a: 0.19  b: 0.56999999999999995
    if (a * 3 == b) {
        cout << "相等" << endl;
    } else {
        cout << a * 3 << endl;  // 0.57
        cout << "no" << endl;   // 输出 no
    }

    if (fabs(a * 3 - b) < 1e-12) {
        cout << "yes" << endl;
    } else {
        cout << "no" << endl;   // 输出 yes
    }
}
```

# swtich 选择结构

```c++
#include <iostream>

using namespace std;

/**
 * switch 语句
 *      实现多路选择
 */

int main() {

    char grade;
    cin >> grade;
    switch (grade) {
        // 这样写会造成 case 穿透
        case 'A':
            cout << "90-100" << endl;
        case 'B':
            cout << "80-89" << endl;
        case 'C':
            cout << "70-79" << endl;
        case 'D':
            cout << "60-69" << endl;
        default:
            cout << "0-59" << endl;
    }

    cout << "---------分割线---------" << endl;

    switch (grade) {
        // 这样写不会造成 case 穿透
        case 'A':
            cout << "90-100" << endl;
            break;
        case 'B':
            cout << "80-89" << endl;
            break;
        case 'C':
            cout << "70-79" << endl;
            break;
        case 'D':
            cout << "60-69" << endl;
            break;
        default:
            cout << "0-59" << endl;
            break;
    }
}
```

# switch 结构例子

```c++
#include <iostream>

using namespace std;

/**
 * switch 语句例子
 *      编写程序，输入月份，输出该月份包含的天数，例如 1 月有 31 天。

 */

int main() {

    int month;

    cin >> month;

    switch (month) {
        case 1:
        case 3:
        case 5:
        case 7:

        case 10:
        case 12:

        case 4:
        case 6:
```

# if 和 switch 结构的比较

* switch 智能以整数相等作为入口判断条件，if 语句可以是任何类型条件，甚至是其它类型值
* switch 语句实现多路分支很自然，if 语句需要嵌套
* 不加 break 语句的 switch 语句是一种变形的分支，有时候很有用
* 各有优缺点，根据实际情况选择合适的结构

# 什么时候使用 if-else？什么时候使用 switch？

* 热量传输，A 的温度 > B 的温度，则从 A 传到 B，小于则从 B 传到 A：if-else
* 水的状态，温度 > 100，为气态：if-else
* 处理交通违章，乱停扣 3 分，罚 200，如果有违章编号时：switch
* 根据距离和费用选择接通工具：if-else (但这是一个更智能的问题，需要更好的算法去解决)
* 汽车根据速度选择档位，大于 25 进 2 档：if-else
* ……

多数情况下还是使用 if-else 较多

# switch 结构的另一个例子

```c++
#include <iostream>

using namespace std;

/**
 * switch 语句例子
 *      编写程序，输入1999-9-9，计算该日期是这一年的第几天。
 */

int main() {
    int year, month, day;
    cout << "请输入年月日:" << endl;

    // 多个输入以空格分割
    cin >> year >> month >> day;
    // 假设输入都是合法的
    int days = day;
    switch (month) {
        case 12:
            days += 31;
        case 11:
            days += 30;
        case 10:
            days += 30;
        case 9:
            days += 31;
        case 8:
            days += 31;
        case 7:
            days += 30;
        case 6:
            days += 31;
        case 5:
            days += 30;
        case 4:
            days += 31;
        case 3:
        {
            if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0) {
                days += 29;
            } else {
                days += 28;
            }
        }
        case 2:
            days += 31;
        case 1:
            break;
        default:
            cout << "请输入正确的月份" << endl;
            break;
    }

    cout << year << "-" << month << "-" << day << ": 总天数是 " << days << " 天" << endl;
}
```
